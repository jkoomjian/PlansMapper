CREATE TABLE `addresses` (
  `username` varchar(32) NOT NULL,
  `lat` varchar(32) NOT NULL,
  `lng` varchar(32) NOT NULL,
  `addr` varchar(512) NOT NULL,
  `city` varchar(512),
  `neighborhood` varchar(512),
  `state` varchar(64),
  `zip` varchar(9),
  PRIMARY KEY (`username`),
  KEY `neighborhood` (`neighborhood`)
);
